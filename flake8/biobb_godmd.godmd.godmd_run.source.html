107 def create_godmdin(self, path: str = None) -> str:
108 """Creates a GOdMD configuration file (godmdin) using the properties file settings"""
109 godmdin_list = []
110  
111 self.output_godmdin_path = path
112  
113 if self.io_dict['in']['input_config_path']:
114 # GOdMD input parameters read from an input godmdin file
115 with open(self.io_dict['in']['input_config_path']) as input_params:
116 for line in input_params:
117 if '=' in line:
118 godmdin_list.append(line.upper())
119 else:
120 # Pre-configured simulation type parameters
121 godmdin_list.append(" TSNAP = 500 ! BioBB GOdMD default params \n")
122 godmdin_list.append(" TEMP = 300 ! BioBB GOdMD default params \n")
123 godmdin_list.append(" SEED = 2525 ! BioBB GOdMD default params \n")
124 godmdin_list.append(" ENER_EVO_SIZE = 20 ! BioBB GOdMD default params \n")
125 godmdin_list.append(" NBLOC = 10000 ! BioBB GOdMD default params \n")
126 godmdin_list.append(" ERRORACCEPTABLE = 1.5 ! BioBB GOdMD default params \n")
127  
128 # Adding the rest of parameters in the config file to the mdin file
129 # if the parameter has already been added replace the value
130 parameter_keys = [parameter.split('=')[0].strip() for parameter in godmdin_list]
131 for k, v in self.godmdin.items():
132 config_parameter_key = str(k).strip().upper()
133 if config_parameter_key in parameter_keys:
134 godmdin_list[parameter_keys.index(config_parameter_key)] = '\t' + config_parameter_key + ' = ' + str(v) + ' ! BioBB property \n'
135 else:
136 godmdin_list.append('\t' + config_parameter_key + ' = '+str(v) + ' ! BioBB property \n')
137  
138 # Writing MD configuration file (mdin)
139 with open(self.output_godmdin_path, 'w') as godmdin:
140  
141 # GOdMDIN parameters added by the biobb_godmd module
142 godmdin.write("!This godmdin file has been created by the biobb_godmd module from the BioBB library \n\n")
143  
144 godmdin.write("&INPUT\n")
145  
146 # MD config parameters
147 for line in godmdin_list:
148 godmdin.write(line)
149  
150 godmdin.write("&END\n")
151  
152 return self.output_godmdin_path
